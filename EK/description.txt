cat ~/.gitconfig
	смотрим что находиться в гит конфиге

git
	может находиться в 3 состояниях
		modified - изменён
			файлы которые поменялись но ещё не были зафиксированы
		staged - индексирован
			изменённый файл в его текущей версии, отмеченный для включения в следующий коммит
		committed - зафиксирован
			сохранён в локальной базе

	config
		--global 
			user.name
			user.email
				задаём имя и имейл в конфигурации Гита чтобы другие разработчики могли видеть кто комитет
	status
		смотрим какие изменения мы сделали в проекте
	add
		добовляем выбранный или все изменения в проект
	log
		смотрим все коммиты которые есть на проекте
		--oneline
			показать в одну строку коммиты 
		--all
			показать все коммиты, из всех веток
		--graph
			показывает в графах коммиты(ответвление)
	show 
		чтобы посмотреть что было изменно в выбранном коммите подставляем hash код
	checkout
		отменяем изменения которые ещё не были добавленны в коммит
		востанавливает удалённый файл
		можно откатить до нужного коммита 
			git checkout "hash коммита" one.txt(указываю путь до файла)
		переключается на другую ветку
		-b (название новой ветки)
	reset 
		удаляем файл из индексированных 
		git reset HEAD (путь до файла который хотим исключить)
	commit 
		фиксируем изменения в гите 
		--amend
			изменяем коммит не залитый на сервер
			--no-edit 
				сообщения коммита не будем изменять
	rm		
		удаляем не нужные файлы из commit'a
		--cached
			git rm --cached (путь до файла)
			удаляем из индексации но не из рабочей директории
	merge test(указываем ветку откуда хотим взять изменения) -m master(указываем ветку куда хотим добавить изменения)
		находимся в ветке master
	remote
		добовлят наше имя в удалённый репазиторий и устанавливает ему имя origin 
		-v
			показывает имя репозитория и адрес
		set-url
			меняем адрес
			git remote set-url origin git@githab.com:.../test.git
	branch
		проврерить ветки локально
		-M main
			указываем имя ветке 
			git branch -M main
	push
		добавляет локальный проект в удалённый репозиторий
		git push origin remot-branch:remote-branch
			изменяем имя ветки если ошиблись в названии
	clone
		колонируем проект

SSH
	необходимо создать и добавить ССШ ключь в корневой директории, нужно сгенерировать два ключа открытый и закрытый ключь
	открытый ключ мы добавим в настройки Гитхаба
	закрытый храним на компе
	cd ~ 
		переходим в папку на компе
	mkdir .ssh
		создаём файл .ssh
	cd .ssh	
		переходим в .ssh
	ssh-keygen -o
		генерируем ssh ключь
	cat id_rsa.pub
		выводим наш публичный ключ
